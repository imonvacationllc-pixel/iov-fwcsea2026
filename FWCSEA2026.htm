        function initializePage() {

            // Create move selector buttons

            const selectorHtml = moves.map((move, index) => `

                <button class="move-btn ${index === 0 ? 'active' : ''}" onclick="selectMove('${move.id}')">

                    <span class="emoji">${move.emoji}</span>

                    <div>${move.name}</div>

                </button>

            `).join('');

 

            document.getElementById('moveSelector').innerHTML = selectorHtml;

 

            // Create move content

            let contentHtml = '';

            moves.forEach((move, index) => {

                contentHtml += createMoveContent(move);

            });

 

            document.getElementById('moveContent').innerHTML = contentHtml;

 

            // Hide all except first

            document.querySelectorAll('.move-content').forEach((el, index) => {

                if (index !== 0) el.classList.remove('active');

            });

        }

 

        function selectMove(moveId) {

            // Hide all content

            document.querySelectorAll('.move-content').forEach(el => el.classList.remove('active'));

           

            // Show selected

            document.getElementById(`move-${moveId}`).classList.add('active');

 

            // Update buttons

            document.querySelectorAll('.move-btn').forEach(btn => btn.classList.remove('active'));

            event.target.closest('.move-btn').classList.add('active');

 

            // Scroll to top

            document.querySelector('.move-selector').scrollIntoView({ behavior: 'smooth' });

        }

 

        // Initialize on page load

        window.addEventListener('DOMContentLoaded', initializePage);
